public class Student{
   private int stuNo;
   private String stuName;
   public void setStuNo(int stuNo){
       this.stuNo = stuNo;
   }
   public int getStuNo(){
       return stuNo;
   }
   public void setStuName(String stuName){
      this.stuName = stuName
   }
   public String getStuName(){
      return stuName;
   }
}
